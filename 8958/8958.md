fun main(){
    val sc = Scanner(System.`in`)

    val b = sc.next()

let num = Int(readLine()!)!
var result : Array<Int> = [] //빈 배열 선언시 자료형 지정

    let input = readLine()!.split(separator: "X").map{String($0).count}.map{$0 * ($0+1) / 2}
    result.append(input.reduce(0,+))
}
result.map{print($0)}
